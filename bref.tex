% Bref is defined to find the last word in a string, and then call
% ibibleref with this last word as a chapter-verse argument, and use
% the rest of the string as a book title. The code is due to David Carlisle
\protected\def\Bref#1{\xBref{}#1 \relax}
\protected\def\BrefChV#1{\yBref{}#1 \relax}
\protected\def\BrefV#1{\zBref{}#1 \relax}

\def\xBref#1#2 #3{%
\ifx\relax#3%
\ibibleverse{\@nameuse{\detokenize{#1}}}(#2)% detokenize needed to deal with encoded strings
\expandafter\xthree
\fi
\xBref{#1#2}#3}

\def\yBref#1#2 #3{% does not display book name, only chapter & verse(s)
\ifx\relax#3%
\ibiblechvs{\@nameuse{\detokenize{#1}}}(#2)% detokenize needed to deal with encoded strings
\expandafter\xthree
\fi
\yBref{#1#2}#3}

\def\zBref#1#2 #3{% does not display book name, only chapter & verse(s)
\ifx\relax#3%
\ibiblevs{\@nameuse{\detokenize{#1}}}(#2)% detokenize needed to deal with encoded strings
\expandafter\xthree
\fi
\zBref{#1#2}#3}
